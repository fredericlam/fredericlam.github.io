# Cancer data analysis example
import pandas as pd
import matplotlib.pyplot as plt

def analyze_cancer_data(data_file):
    """
    Analyze cancer incidence data
    """
    df = pd.read_csv(data_file)
    results = df.groupby('cancer_type').agg({
        'incidence': 'mean',
        'mortality': 'sum'
    })
    return results

# Load and analyze data
data = analyze_cancer_data('globocan_2024.csv')
print(data.head())
